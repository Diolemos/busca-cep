                    from typing import List
                    
                    def encontreNumerofaltante(array: List[int]) -> List[int]:
                        answer = [numero + 1 for numero in range(len(array) - 1) if numero + 1 != array[numero]]
                        return answer
                    
                    print(encontreNumerofaltante([1, 2, 3, 6, 7, 8]))

                    const findNumWhichSumEqualsZero = (array) => {
                      let counter = 0;
                      let answer = [];
                    
                      for (let i = 0; i <= array.length - 3; i++) {
                        if (array[i] + array[i + 1] + array[i + 2] === 0) {
                          counter++;
                          answer.push([array[i], array[i + 1], array[i + 2]]);
                        }
                      }
                    
                      counter
                        ? console.log(`A sequência apresenta ${counter} combinações que resultam em zero`)
                        : console.log(`A sequência não apresenta combinações que resultam em zero`);
                    
                      return answer;
                    };
                    
                    console.log(findNumWhichSumEqualsZero([-1, 0, 1, 2, -1, -4]));

  • 3- Escreva um algoritmo para encontrar três números consecutivos de um array que somados o resultado é igual a um outro número fornecido: Input : (array(2, 7, 7, 1, 8, 2, 7, 8, 7), 16)); Output : array( [0] => 2 + 7 + 7 = 16, [1] => 7 + 1 + 8 = 16 ); 

    
                    function encontreTresNumeros(array, target) {
                        let resposta = [];
                      
                        for (let i = 0; i <= array.length - 3; i++) {
                          if (array[i] + array[i + 1] + array[i + 2] === target) {
                            resposta.push([array[i], array[i + 1], array[i + 2]]);
                          }
                        }
                      
                        return resposta;
                      }
                      
                      const resultado = encontreTresNumeros([2, 7, 7, 1, 8, 2, 7, 8, 7], 16);
