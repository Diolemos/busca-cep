                    def encontreNumerofaltante(array: List[int]) -> List[int]:

                    function findSingleNumber(array) {
                        let resposta = [];
                        let singleNumber;
                      
                        array.forEach(function (currentValue) {
                          let counter = 0;
                      
                          for (let i = 0; i < array.length; i++) {
                            if (array[i] === currentValue) {
                              counter++;
                            }
                          }
                      
                          if (counter === 1) {
                            singleNumber = currentValue;
                          }
                        });
                      
                        return [array, singleNumber];
                      }
                      
                      const resultado4 = findSingleNumber([5, 3, 4, 3, 4]);
                      console.log(resultado4);

                        function findSingleNumberWithThreeRepeats(array) {
                            let countMap = {};
                          
                            array.forEach(function (num) {
                              countMap[num] = (countMap[num] || 0) + 1;
                            });
                          
                            let singleNumber;
                          
                            for (let num in countMap) {
                              if (countMap[num] !== 3) {
                                singleNumber = parseInt(num);
                                break;
                              }
                            }
                          
                            return [array, singleNumber];
                          }
                          
                          const resultado5 = findSingleNumberWithThreeRepeats([5, 3, 4, 3, 5, 5, 3]);

                        function findFibonacciSequence(num) {
                            let resposta = [];
                          
                            if (num === 1) {
                              resposta.push(1);
                            } else if (num === 2) {
                              resposta = [1, 1];
                            } else {
                              resposta = [1, 1];
                          
                              for (let i = 2; i < num; i++) {
                                resposta.push(resposta[i - 2] + resposta[i - 1]);
                              }
                            }
                          
                            return resposta;
                          }
                          
                          const resultado6 = findFibonacciSequence(6);
